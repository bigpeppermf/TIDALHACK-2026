\section*{The Power Rule combined with the Chain Rule (Generalized Power Rule)}
If $n$ is a real number then
\[ \frac{d}{dx} (g(x))^n = n(g(x))^{n-1} g'(x) \quad \text{or} \quad \frac{d}{dx} (u)^n = n u^{n-1} \frac{du}{dx} \]

\section*{Example 1 Differentiate the following functions.}
\begin{enumerate}
    \item $y = (1+2x)^{10} = (1+2x)(1+2x) \cdots (1+2x)$
    \begin{align*}
        y' &= 10(1+2x)^9 \cdot 2
    \end{align*}
    \item $y = \frac{1}{(1+2x)^{2024}} = (1+2x)^{-2024}$
    \begin{align*}
        y' &= -2024(1+2x)^{-2025} \cdot 2
    \end{align*}
    \item $F(x) = \sqrt[3]{x^2+2x-3} = (x^2+2x-3)^{\frac{1}{3}}$
    \begin{align*}
        F'(x) &= \frac{1}{3}(x^2+2x-3)^{-\frac{2}{3}}(2x+2)
    \end{align*}
    \item $f(x) = \frac{1}{\sqrt[4]{x^2+x+1}} = (x^2+x+1)^{-\frac{1}{4}}$
    \begin{align*}
        f'(x) &= -\frac{1}{4}(x^2+x+1)^{-\frac{5}{4}}(2x+1)
    \end{align*}
    \item $y = (2x+1)^5(x^3-x+1)^4$
    \begin{align*}
        y' &= ((2x+1)^5)'(x^3-x+1)^4 + (2x+1)^5((x^3-x+1)^4)' \\
        &= 5(2x+1)^4 \cdot 2 (x^3-x+1)^4 + (2x+1)^5 \cdot 4 (x^3-x+1)^3 (3x^2-1)
    \end{align*}
    \item $g(x) = \sin 2x$
    \[ x \longrightarrow 2x \stackrel{\sin}{\longrightarrow} \sin(2x) \]
    \begin{align*}
        g'(x) &= \cos(2x) \cdot 2
    \end{align*}
    \item $f(x) = \sin \frac{1}{x}$
    \[ x \longrightarrow \frac{1}{x} \stackrel{\sin}{\longrightarrow} \sin \frac{1}{x} \]
    \begin{align*}
        f' &= \cos\left(\frac{1}{x}\right) \cdot (-x^{-2})
    \end{align*}
\end{enumerate}
